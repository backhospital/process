CREATE TABLE `offer` (
	`idx` INT(11) NOT NULL AUTO_INCREMENT,
	`brandname` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`productname` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`offerprice` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`offerdes` VARCHAR(500) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`offerimg` VARCHAR(500) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`offerthumbimg` VARCHAR(500) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`offerdate` TIMESTAMP NULL DEFAULT current_timestamp(),
	`verify` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	PRIMARY KEY (`idx`) USING BTREE
)
COLLATE='utf8mb4_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=0
;
